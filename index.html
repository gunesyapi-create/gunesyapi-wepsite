""" JARVIS Windows - Gerçek Zamanlı Sesli Asistan Basit ve çalışan versiyon """ import asyncio import json import datetime import traceback import os import signal import sys from pathlib import Path from typing import Optional # Ses kütüphanesi import pyaudio import wave # Google Gemini from google import genai from google.genai import types # ------------------------------------------------------------ # KONFİGÜRASYON # ------------------------------------------------------------ CONFIG_FILE = Path("config.json") MEMORY_FILE = Path("memory.json") FORMAT = pyaudio.paInt16 CHANNELS = 1 SEND_SAMPLE_RATE = 16000 # Mikrofon RECV_SAMPLE_RATE = 24000 # Hoparlör CHUNK_SIZE = 1024 def load_config(): if CONFIG_FILE.exists(): return json.loads(CONFIG_FILE.read_text(encoding="utf-8")) return {"gemini_api_key": "", "voice": "Charon"} def save_config(config): CONFIG_FILE.write_text(json.dumps(config, indent=2), encoding="utf-8") def load_memory(): if MEMORY_FILE.exists(): return json.loads(MEMORY_FILE.read_text(encoding="utf-8")) return {} def save_memory(memory): MEMORY_FILE.write_text(json.dumps(memory, indent=2), encoding="utf-8") def update_memory(key: str, value: str, category: str = "notes"): mem = load_memory() if category not in mem: mem[category] = {} mem[category][key] = {"value": value, "updated": datetime.datetime.now().isoformat()} save_memory(mem) return f"'{key}' kaydedildi." def delete_memory(key: str = None, match_text: str = None): mem = load_memory() deleted = False for cat in list(mem.keys()): for k in list(mem[cat].keys()): if key and k == key: del mem[cat][k] deleted = True elif match_text and match_text.lower() in k.lower(): del mem[cat][k] deleted = True if deleted: save_memory(mem) return "Hafızadan silindi." return "Kayıt bulunamadı." def format_memory_for_prompt(): mem = load_memory() if not mem: return "" lines = ["[KALICI HAFIZA]"] for cat, items in mem.items(): lines.append(f"\n{cat.upper()}:") for key, data in items.items(): lines.append(f" - {key}: {data.get('value', '')}") return "\n".join(lines) # ------------------------------------------------------------ # SİSTEM PROMPT'U # ------------------------------------------------------------ SYSTEM_PROMPT = """Sen JARVIS'sin — Windows'ta çalışan kişisel AI asistanı. Türkçe konuş. Kısa ve net yanıtlar ver. Kibar, yardımsever ve profesyonel ol. Kullanabileceğin araçlar (function calling): - get_time: Şu anki saati söyler - get_date: Bugünün tarihini söyler - save_memory: Kullanıcı hakkında bilgi kaydeder (kategori, anahtar, değer) - delete_memory: Kayıtlı bilgiyi siler Araçları gerektiğinde çağır, asla taklit etme. Kullanıcıya doğrudan yardımcı ol.""" # ------------------------------------------------------------ # ARAÇ TANIMLARI # ------------------------------------------------------------ TOOLS = [ { "name": "get_time", "description": "Şu anki saati döndürür", "parameters": { "type": "OBJECT", "properties": {} } }, { "name": "get_date", "description": "Bugünün tarihini döndürür", "parameters": { "type": "OBJECT", "properties": {} } }, { "name": "save_memory", "description": "Kullanıcı hakkında önemli bilgiyi kalıcı belleğe kaydeder", "parameters": { "type": "OBJECT", "properties": { "key": {"type": "STRING", "description": "Bilginin anahtarı (örn: 'name')"}, "value": {"type": "STRING", "description": "Kaydedilecek değer"}, "category": {"type": "STRING", "description": "Kategori (identity/preferences/projects/notes)"} }, "required": ["key", "value"] } }, { "name": "delete_memory", "description": "Kalıcı hafızadan bilgi siler", "parameters": { "type": "OBJECT", "properties": { "key": {"type": "STRING", "description": "Silinecek anahtar"}, "match_text": {"type": "STRING", "description": "Alternatif: eşleşen metin"} } } } ] # ------------------------------------------------------------ # JARVIS ANA SINIFI # ------------------------------------------------------------ class JarvisAsistan: def __init__(self): self.config = load_config() self.client = None self.session = None self.audio_in_queue = None self.out_queue = None self.is_speaking = False self.running = True self.pya = pyaudio.PyAudio() # API anahtarı kontrolü if not self.config.get("gemini_api_key"): print("❌ Lütfen config.json dosyasına gemini_api_key ekleyin!") sys.exit(1) self.client = genai.Client( api_key=self.config["gemini_api_key"], http_options={"api_version": "v1alpha"} ) def _build_config(self): now = datetime.datetime.now() memory_text = format_memory_for_prompt() system_text = f"""[ŞU ANKİ ZAMAN] {now.strftime('%A, %d %B %Y — %H:%M')} {memory_text} {SYSTEM_PROMPT}""" return types.LiveConnectConfig( response_modalities=["AUDIO"], output_audio_transcription={}, input_audio_transcription={}, system_instruction=system_text, tools=[{"function_declarations": TOOLS}], speech_config=types.SpeechConfig( voice_config=types.VoiceConfig( prebuilt_voice_config=types.PrebuiltVoiceConfig( voice_name=self.config.get("voice", "Charon") ) ) ), ) async def _execute_tool(self, fc): name = fc.name args = dict(fc.args or {}) print(f"\n🔧 {name}({args})") try: if name == "get_time": now = datetime.datetime.now() result = f"Saat {now.strftime('%H:%M')}" elif name == "get_date": now = datetime.datetime.now() result = f"Bugün {now.strftime('%d %B %Y')}" elif name == "save_memory": key = args.get("key", "") value = args.get("value", "") category = args.get("category", "notes") if key and value: result = update_memory(key, value, category) else: result = "Hata: key ve value gerekli" elif name == "delete_memory": result = delete_memory( args.get("key", ""), args.get("match_text", "") ) else: result = f"Bilinmeyen araç: {name}" except Exception as e: result = f"Hata: {e}" traceback.print_exc() print(f"📤 Sonuç: {result[:100]}") return types.FunctionResponse( id=fc.id, name=name, response={"result": result} ) async def _send_realtime(self): """Mikrofondan gelen sesi gönder""" while self.running: try: msg = await self.out_queue.get() await self.session.send_realtime_input(media=msg) except Exception as e: print(f"Mikrofon gönderme hatası: {e}") break async def _listen_audio(self): """Mikrofonu dinle""" print("🎤 Mikrofon aktif...") stream = await asyncio.to_thread( self.pya.open, format=FORMAT, channels=CHANNELS, rate=SEND_SAMPLE_RATE, input=True, frames_per_buffer=CHUNK_SIZE ) try: while self.running: data = await asyncio.to_thread( stream.read, CHUNK_SIZE, exception_on_overflow=False ) if not self.is_speaking: await self.out_queue.put({"data": data, "mime_type": "audio/pcm"}) except Exception as e: print(f"Mikrofon hatası: {e}") finally: stream.close() async def _receive_audio(self): """Gemini'den gelen yanıtları işle""" print("👂 Alım başladı...") try: async for response in self.session.receive(): # Ses verisi geldiğinde if response.data: self.audio_in_queue.put_nowait(response.data) if response.server_content: sc = response.server_content # JARVIS'in konuşması if sc.output_transcription and sc.output_transcription.text: self.is_speaking = True text = sc.output_transcription.text.strip() if text: print(f"\n🗣️ JARVIS: {text}") # Kullanıcının konuşması (transkript) if sc.input_transcription and sc.input_transcription.text: text = sc.input_transcription.text.strip() if text: print(f"\n👤 Siz: {text}") # Konuşma bitti if sc.turn_complete: self.audio_in_queue.put_nowait(None) # Ses çalmayı bitir self.is_speaking = False print("\n🎧 Dinliyorum...") # Araç çağrıları if response.tool_call: fn_responses = [] for fc in response.tool_call.function_calls: fr = await self._execute_tool(fc) fn_responses.append(fr) await self.session.send_tool_response(function_responses=fn_responses) except Exception as e: print(f"Alım hatası: {e}") raise async def _play_audio(self): """Sesi çal""" print("🔊 Ses çalma başladı...") stream = await asyncio.to_thread( self.pya.open, format=FORMAT, channels=CHANNELS, rate=RECV_SAMPLE_RATE, output=True ) try: while self.running: chunk = await self.audio_in_queue.get() if chunk is None: # Turn complete - ses bitti continue await asyncio.to_thread(stream.write, chunk) except Exception as e: print(f"Ses çalma hatası: {e}") finally: stream.close() async def run(self): """Ana döngü""" print("\n" + "="*50) print("🤖 JARVIS Asistan Başlatılıyor...") print("="*50 + "\n") while self.running: try: config = self._build_config() async with self.client.aio.live.connect( model="models/gemini-2.0-flash-exp", # Ses destekli model config=config ) as session: self.session = session self.audio_in_queue = asyncio.Queue() self.out_queue = asyncio.Queue(maxsize=10) print("✅ Bağlantı kuruldu!") print("🎙️ Konuşmaya başlayabilirsiniz...\n") async with asyncio.TaskGroup() as tg: tg.create_task(self._send_realtime()) tg.create_task(self._listen_audio()) tg.create_task(self._receive_audio()) tg.create_task(self._play_audio()) except Exception as e: print(f"\n⚠️ Bağlantı hatası: {e}") print("🔄 5 saniye sonra yeniden bağlanılıyor...") await asyncio.sleep(5) def stop(self): self.running = False # ------------------------------------------------------------ # MAIN # ------------------------------------------------------------ def main(): # Windows'ta asyncio sorunları için if sys.platform == "win32": asyncio.set_event_loop_policy(asyncio.WindowsSelectorEventLoopPolicy()) asistan = JarvisAsistan() # Ctrl+C için sinyal işleyici def signal_handler(sig, frame): print("\n\n🛑 Kapatılıyor...") asistan.stop() sys.exit(0) signal.signal(signal.SIGINT, signal_handler) try: asyncio.run(asistan.run()) except KeyboardInterrupt: print("\n👋 Görüşmek üzere!") if __name__ == "__main__": main()
